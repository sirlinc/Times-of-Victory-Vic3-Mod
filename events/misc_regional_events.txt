namespace = war_score_events

war_score.1 = { #Capitulation
	type = country_event
	placement = root
	hidden = yes
	
	trigger = {
		is_at_war = yes
		is_ai = yes
		is_country_alive = yes
		has_war_exhaustion = { target = c:root value > 50 }
		any_scope_state = {
			is_capital = yes
			devastation > 50
		}
	}

	immediate = {
		add_modifier = {
             name = capital_capturated
             months = 100
        }
	}
}

war_score.2 = { #Remove capital modifier
	type = country_event
	#placement = root
	hidden = yes
	
	trigger = {
		is_at_war = no
		has_modifier = capital_capturated
	}

	immediate = {
		remove_modifier = capital_capturated
	}
	
}
