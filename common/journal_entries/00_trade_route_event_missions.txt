je_build_local_arms_industry = {
	icon = "gfx/interface/icons/event_icons/event_trade.dds"

	complete = {
		any_scope_state = {
			has_active_building = building_arsenal
		}
	}

	on_complete = {
		ig:ig_expansionist = {
			add_modifier = {
				name = ig_domestic_weapons_manufacturing
				months = short_modifier_time
			}
		}
	}


	on_timeout = {
		ig:ig_expansionist = {
			add_modifier = {
				name = ig_unreliable_weapon_imports
				months = short_modifier_time
			}
		}
	}

	timeout = 600

	weight = 200

	transferable = no
}

je_set_up_grain_import = {
	icon = "gfx/interface/icons/event_icons/event_trade.dds"

	complete = {
		capital.market = {
			any_trade_route = {
				importer.owner = root
				goods = g:grain
			}
		}
	}

	on_complete = {
		scope:target = {
			add_modifier = {
				name = ig_secured_desired_imports
				months = normal_modifier_time
			}
		}
	}


	on_timeout = {
		scope:target = {
			add_modifier = {
				name = ig_refused_desired_imports
				months = normal_modifier_time
			}
		}
	}

	timeout = 600

	weight = 200

	transferable = no
}
