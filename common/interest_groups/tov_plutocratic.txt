ig_plutocracy = {
	color = rgb { 173 66 12 }
	texture = "gfx/interface/icons/ig_icons/plutocratic.dds"
	
	layer = "revolution_dynamic_industrialists"
	index = 2
	
	ideologies = {
		ideology_plutocratic	
		ideology_stratocratic
	}

	enable = { always = yes }

	set_variable = { name = techonology }
	
	traits = {}	
	
	on_enable = {}
	on_disable = {}	
	on_character_ig_membership = {}
	
	pop_potential = {
		OR = {
			is_pop_type = aristocrats
			is_pop_type = capitalists
			NOR = {
				pop_employment_building_group = bg_agriculture
				pop_employment_building_group = bg_mining
				pop_employment_building_group = bg_ranching
				pop_employment_building_group = bg_plantations
			}
		}
	}

	pop_weight = {
		value = 0
	
		add = {
			desc = "POP_CAPITALISTS"
			if = {
				limit = { 
					is_pop_type = capitalists
				}
				value = 10
			}
		}	
		
		add = {
			desc = "POP_OFFICERS_SOL_11"
			if = {
				limit = { 
					is_pop_type = officers
					standard_of_living >= 11
				}
				value = 6.25
			}
		}	
		
		add = {
			desc = "POP_ARISTOCRATS"
			if = {
				limit = { 
					is_pop_type = aristocrats
				}
				value = 1.5
			}
		}	
		
		add = {
			desc = "POP_URBAN_WORKERS"
			if = {
				limit = { 
					is_employed = yes
					NOR = {
						pop_employment_building_group = bg_agriculture
						pop_employment_building_group = bg_mining
						pop_employment_building_group = bg_ranching
						pop_employment_building_group = bg_plantations
					}
				}
				value = 2.85
			}
		}
		
		add = {
			desc = "POP_RURAL_WORKERS"
			if = {
				limit = { 
					OR = {
						pop_employment_building_group = bg_agriculture
						pop_employment_building_group = bg_mining
						pop_employment_building_group = bg_ranching
						pop_employment_building_group = bg_plantations
					}
				}
				value = -2.0
			}
		}
		
		#SOL Upper Strata
		if = {
			limit = {
				OR = {
					is_pop_type = capitalists 
					is_pop_type = aristocrats
					is_pop_type = shopkeepers
				}
				standard_of_living >= 15
				standard_of_living < 20
			}
			multiply = {
				desc = "SOL_15_20"
				value = 1.0
			}
		}
		
		if = {
			limit = {
				OR = {
					is_pop_type = capitalists 
					is_pop_type = aristocrats
					is_pop_type = shopkeepers
				}
				standard_of_living >= 20
				standard_of_living < 25
			}
			multiply = {
				desc = "SOL_20_25"
				value = 1.15
			}
		}
		
		if = {
			limit = {
				OR = {
					is_pop_type = capitalists 
					is_pop_type = aristocrats
					is_pop_type = shopkeepers
				}
				standard_of_living >= 25
				standard_of_living < 30
			}
			multiply = {
				desc = "SOL_25_30"
				value = 1.25
			}
		}
		
		if = {
			limit = {
				OR = {
					is_pop_type = capitalists 
					is_pop_type = aristocrats
					is_pop_type = shopkeepers
				}
				standard_of_living >= 30
				standard_of_living < 35
			}
			multiply = {
				desc = "SOL_30_35"
				value = 1.5
			}
		}
		
		if = {
			limit = {
				OR = {
					is_pop_type = capitalists 
					is_pop_type = aristocrats
					is_pop_type = shopkeepers
				}
				standard_of_living >= 35
				standard_of_living < 40
			}
			multiply = {
				desc = "SOL_35_40"
				value = 1.75
			}
		}
		
		if = {
			limit = {
				OR = {
					is_pop_type = capitalists 
					is_pop_type = aristocrats
					is_pop_type = shopkeepers
				}
				standard_of_living >= 40
				standard_of_living < 45
			}
			multiply = {
				desc = "SOL_40_45"
				value = 1.85
			}
		}
		
		multiply = {
			desc = "LEADER_POPULARITY"
			scope:interest_group = {
				leader = {
					value = popularity
					multiply = 0.0025
					add = 1
					max = 1.25
					min = 0.75
				}
			}
		}		

		multiply = {
			desc = "decentralized_power"	
			value = 1.0
			
			if = {
				limit = { 
					owner = {
						is_country_type = decentralized
					}
				}
				value = 0.0
			}		
		}
		if = {
			limit = {
				scope:interest_group = {
					is_in_government = yes
				}
			}
			multiply = {
				desc = "IN_GOVERNMENT_ATTRACTION"
				value = 1
				add = scope:interest_group.modifier:interest_group_in_government_attraction_mult
				min = 0
			}
		}
		
		multiply = {
			desc = "NO_JOIN_STOCK_COMPANY"	
			value = 1.0
			
			if = {
				limit = { 
					owner = {
						NOT = { has_technology_researched = join_stock_company }
					}
				}
				value = 0.65
			}		
		}
	}

	monarch_weight = {
		value = 0.25
	}	
	
	commander_weight = {
		value = 1.0
	}	
	
	noble_chance = {
		value = 0.1
	}
	
	female_commander_chance = {	
		value = 0.0
	}
	
	female_politician_chance = {
		value = 0.0

		if = {
			limit = {
				owner = {
					has_law = law_type:law_women_own_property
				}
				
			}
			add = {
				value = 0.1
			}
		}

		if = {
			limit = {
				owner = {
					has_law = law_type:law_women_in_the_workplace
				}
				
			}
			add = {
				value = 0.1
			}
		}

		if = {
			limit = {
				owner = {
					has_law = law_type:law_womens_suffrage
				}
				
			}
			add = {
				value = 0.1
			}
		}
	}
	
	# The chance that a commander belonging to this IG takes over leadership when it changes
	# scope:character is the most popular commander in the IG
	commander_leader_chance = {
		value = 0.25
		if = {
			limit = {
				exists = scope:character
			}
			multiply = {
				value = 1
				add = scope:character.popularity
				min = 0
			}
		}
	}
}
