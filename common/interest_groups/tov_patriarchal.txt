ig_patriarchal = {
	color = rgb{ 124 61 0 }
	texture = "gfx/interface/icons/ig_icons/patriarchy.dds"
	
	ideologies = { ideology_patriarchal }

	enable = {
		always = yes
	}
	
	traits = {}	
	
	on_enable = {}
	on_disable = {}	
	on_character_ig_membership = {}
	
	pop_potential = {
		strata >= poor
	}

	pop_weight = {
		value = 0
	
		add = {
			desc = "POP_STRATA_POOR"
			if = {
				limit = { 
					strata = poor
				}
				value = 90
			}
		}	
		
		add = {
			desc = "POP_STRATA_MIDDLE"
			if = {
				limit = { 
					strata = middle
					NOT = { is_pop_type = clergymen }
				}
				value = 110
			}
		}	
		
		add = {
			desc = "POP_STRATA_RICH"
			if = {
				limit = { 
					strata = middle
				}
				value = 110
			}
		}	
		
		add = {
			desc = "POP_CLERGYMEN"
			if = {
				limit = { 
					is_pop_type = clergymen
					NOT = { has_technology_researched = feminism }
				}
				value = 120
			}
		}	
		
		add = {
			desc = "POP_CLERGYMEN_FEMINISM"
			if = {
				limit = { 
					is_pop_type = clergymen
					has_technology_researched = feminism
				}
				value = 60
			}
		}	
		
		add = {
			desc = "POP_OFFICERS"
			if = {
				limit = { 
					is_pop_type = officers
				}
				value = 60
			}
		}	
		
		multiply = {
			desc = "LEADER_POPULARITY"
			scope:interest_group = {
				leader = {
					value = popularity
					multiply = 0.0025
					add = 1
					max = 1.25
					min = 0.75
				}
			}
		}		

		multiply = {
			desc = "decentralized_power"	
			value = 1.0
			
			if = {
				limit = { 
					owner = {
						is_country_type = decentralized
					}
				}
				value = 0.0
			}		
		}
		if = {
			limit = {
				scope:interest_group = {
					is_in_government = yes
				}
			}
			multiply = {
				desc = "IN_GOVERNMENT_ATTRACTION"
				value = 1
				add = scope:interest_group.modifier:interest_group_in_government_attraction_mult
				min = 0
			}
		}
	}

	monarch_weight = {
		value = 0.25
	}	
	
	commander_weight = {
		value = 1.0
	}	
	
	noble_chance = {
		value = 0.1
	}
	
	female_commander_chance = {	
		value = 0.0
	}
	
	female_politician_chance = {
		value = 0.0

		if = {
			limit = {
				owner = {
					has_law = law_type:law_women_own_property
				}
				
			}
			add = {
				value = 0.1
			}
		}

		if = {
			limit = {
				owner = {
					has_law = law_type:law_women_in_the_workplace
				}
				
			}
			add = {
				value = 0.1
			}
		}

		if = {
			limit = {
				owner = {
					has_law = law_type:law_womens_suffrage
				}
				
			}
			add = {
				value = 0.1
			}
		}
	}
	
	# The chance that a commander belonging to this IG takes over leadership when it changes
	# scope:character is the most popular commander in the IG
	commander_leader_chance = {
		value = 0.25
		if = {
			limit = {
				exists = scope:character
			}
			multiply = {
				value = 1
				add = scope:character.popularity
				min = 0
			}
		}
	}
}
