ig_nationalism = { #patriotic
	color = rgb { 33 102 66 }
	texture = "gfx/interface/icons/ig_icons/nationalist.dds"
	
	layer = "revolution_dynamic_armed_forces"
	index = 0
	
	ideologies = { ideology_patriotic }

	enable = { always = yes }
	
	traits = {}	
	
	set_variable = { name = techonology }

	on_enable = {}
	on_disable = {}	
	on_character_ig_membership = {}
	
	pop_potential = {
		OR = {
			pop_has_primary_culture = yes
			pop_is_discriminated = no
		}
	}

	pop_weight = {
		value = 0.25
	
		add = {
			desc = "PRIMARY_CULTURE"
			value = 0
			if = {
				limit = { pop_has_primary_culture = yes }
				value = 6.5
			}
		}	
		
		add = {
			desc = "IS_NOT_DISCRIMINATED"
			value = 0
			if = {
				limit = { pop_is_discriminated = no }
				value = 1.6
			}
		}	
		
		add = {
			desc = "RELIGION_IS_ACCEPTED"
			value = 0
			if = {
				limit = { religion_accepted = yes }
				value = 2.75
			}
		}
		
		add = {
			desc = "IG_EMPLOYED_MIC"
			value = 0
			if = {
				limit = {
					OR = {
						pop_employment_building = building_automotive_firms
						pop_employment_building = building_arsenal
						pop_employment_building = building_munitions_plants
						pop_employment_building = building_shipyards
					}
				}
				value = 3.5
			}
		}
		
		add = {
			desc = "POP_OFFICERS"
			if = {
				limit = { is_pop_type = officers }
				value = 8.5
			}
		}
		
		add = {
			desc = "POP_SERVICEMEN"
			if = {
				limit = { is_pop_type = servicemen }
				value = 8.5
			}
		}
		
		#At War
		if = {
			limit = {
				owner = { 
					is_at_war = yes
				}
			}
			multiply = {
				desc = "COUNTRY_IS_AT_WAR" 
				value = 1.15
			}
		}
		
		multiply = {
			desc = "LEADER_POPULARITY"
			scope:interest_group = {
				leader = {
					value = popularity
					multiply = 0.0025
					add = 1
					max = 1.25
					min = 0.75
				}
			}
		}		

		multiply = {
			desc = "decentralized_power"	
			value = 1.0
			
			if = {
				limit = { 
					owner = {
						is_country_type = decentralized
					}
				}
				value = 0.0
			}		
		}
		if = {
			limit = {
				scope:interest_group = {
					is_in_government = yes
				}
			}
			multiply = {
				desc = "IN_GOVERNMENT_ATTRACTION"
				value = 1
				add = scope:interest_group.modifier:interest_group_in_government_attraction_mult
				min = 0
			}
		}
		
		multiply = {
			desc = "NO_THE_NATION_STATE"	
			value = 1.0
			
			if = {
				limit = { 
					owner = {
						NOT = { has_technology_researched = the_nation_state }
					}
				}
				value = 0.5
			}		
		}
		
		multiply = {
			desc = "NO_NATIONAL_MYTHOS"	
			value = 1.0
			
			if = {
				limit = { 
					owner = {
						NOT = { has_technology_researched = national_mythos }
					}
				}
				value = 0.25
			}		
		}
	}

	monarch_weight = {
		value = 0.25
	}	
	
	commander_weight = {
		value = 1.0
	}	
	
	noble_chance = {
		value = 0.1
	}
	
	female_commander_chance = {	
		value = 0.0
	}
	
	female_politician_chance = {
		value = 0.0

		if = {
			limit = {
				owner = {
					has_law = law_type:law_women_own_property
				}
				
			}
			add = {
				value = 0.1
			}
		}

		if = {
			limit = {
				owner = {
					has_law = law_type:law_women_in_the_workplace
				}
				
			}
			add = {
				value = 0.1
			}
		}

		if = {
			limit = {
				owner = {
					has_law = law_type:law_womens_suffrage
				}
				
			}
			add = {
				value = 0.1
			}
		}
	}
	
	# The chance that a commander belonging to this IG takes over leadership when it changes
	# scope:character is the most popular commander in the IG
	commander_leader_chance = {
		value = 0.25
		if = {
			limit = {
				exists = scope:character
			}
			multiply = {
				value = 1
				add = scope:character.popularity
				min = 0
			}
		}
	}
}
