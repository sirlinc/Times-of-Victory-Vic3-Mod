ig_elitist = {
	color = hsv{ 0.65 0.54 0.52 }
	texture = "gfx/interface/icons/ig_icons/elitism.dds"
	
	layer = "revolution_dynamic_landowners"
	index = 4
	
	ideologies = {
		ideology_stratocratic
	}

	enable = {
		always = yes
	}
	
	traits = {}	
	
	on_enable = {}
	on_disable = {}	
	on_character_ig_membership = {}
	
	pop_potential = { 
		OR = {
			is_pop_type = aristocrats
			is_pop_type = clergymen
			is_pop_type = capitalists
		}
	}

	pop_weight = {
		value = 0
		
		add = {
			desc = "POP_ARISTOCRATS_SOL_19"
			if = {
				limit = { 
					is_pop_type = aristocrats
					standard_of_living > 19
				}
				value = 200
			}
		}	
		
		add = {
			desc = "POP_ARISTOCRATS_SOL_15"
			if = {
				limit = { 
					is_pop_type = aristocrats
					standard_of_living > 15
					standard_of_living <= 19
				}
				value = 150
			}
		}	
		
		add = {
			desc = "POP_ARISTOCRATS_SOL_10"
			if = {
				limit = { 
					is_pop_type = aristocrats
					standard_of_living <= 15
				}
				value = 60
			}
		}	
		
		add = {
			desc = "POP_CLERGYMEN_SOL_15"
			if = {
				limit = { 
					is_pop_type = clergymen
					standard_of_living > 15
				}
				value = 90
			}
		}	

		add = {
			desc = "POP_CLERGYMEN_SOL_10"
			if = {
				limit = { 
					is_pop_type = clergymen
					standard_of_living > 10
					standard_of_living <= 15
				}
				value = 45
			}
		}	
		
		add = {
			desc = "POP_FARMERS"
			if = {
				limit = { 
					is_pop_type = farmers
				}
				value = 30
			}
		}		

		multiply = {
			desc = "LEADER_POPULARITY"
			scope:interest_group = {
				leader = {
					value = popularity
					multiply = 0.0025
					add = 1
					max = 1.25
					min = 0.75
				}
			}
		}		

		multiply = {
			desc = "IG_IS_MARGINALIZED"
			value = 1.0
			
			scope:interest_group = {
				if = {
					limit = {
						is_marginal = yes
					}
					value = 0.5
				}
			}
		}
		
		if = {
			limit = {
				scope:interest_group = {
					is_in_government = yes
				}
			}
			multiply = {
				desc = "IN_GOVERNMENT_ATTRACTION"
				value = 1
				add = scope:interest_group.modifier:interest_group_in_government_attraction_mult
				min = 0
			}
		}
	}

	monarch_weight = {
		value = 0.45
	}	
	
	commander_weight = {
		value = 1.0
	}	
	
	noble_chance = {
		value = 0.67
	}
	
	female_commander_chance = {	
		value = 0.0
	}
	
	female_politician_chance = {
		value = 0.0
	}
	
	# The chance that a commander belonging to this IG takes over leadership when it changes
	# scope:character is the most popular commander in the IG
	commander_leader_chance = {
		value = 0.25
		if = {
			limit = {
				exists = scope:character
			}
			multiply = {
				value = 1
				add = scope:character.popularity
				min = 0
			}
		}
	}
}
