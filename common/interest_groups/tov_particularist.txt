ig_particularist = {
	color = rgb { 71 105 68 }
	texture = "gfx/interface/icons/ig_icons/particularism.dds" #
	
	ideologies = { ideology_particularist }

	enable = {
		always = yes
	}
	
	traits = {}	
	
	on_enable = {}
	on_disable = {}	
	on_character_ig_membership = {}
	
	pop_potential = {
		NOR = {
			is_pop_type = aristocrats
			is_pop_type = capitalists
		}
		OR = {
			pop_employment_building_group = bg_manufacturing
			pop_employment_building_group = bg_service
			pop_employment_building_group = bg_urban_facilities
			pop_employment_building_group = bg_government
		}
	}

	pop_weight = {
		value = 1
	
		add = {
			desc = "POP_EMPLOYMENT"
			if = {
				limit = { 
					NOR = {
						is_pop_type = aristocrats
						is_pop_type = capitalists
					}
					OR = {
						pop_employment_building_group = bg_manufacturing
						pop_employment_building_group = bg_service
						pop_employment_building_group = bg_urban_facilities
						pop_employment_building_group = bg_government
					}
				}
				value = 100
			}
		}
		
		multiply = {
			desc = "POP_DISCRIMINATED"
			if = {
				limit = { 
					pop_is_discriminated = yes
				}
				value = 1.25
			}
		}
		
		#SoL
		multiply = {
			desc = "SOL_1_5"
			if = { 
				limit = {
					standard_of_living >= 1
					standard_of_living < 5
				}
				value = 1.75
			}
		}	
		multiply = {
			desc = "SOL_5_10"
			if = { 
				limit = {
					standard_of_living >= 1
					standard_of_living < 5
				}
				value = 1.5
			}
		}	
		multiply = {
			desc = "SOL_10_15"
			if = { 
				limit = {
					standard_of_living >= 10
					standard_of_living < 15
				}
				value = 1.25
			}
		}
		multiply = {
			desc = "SOL_20_25"
			if = { 
				limit = {
					standard_of_living >= 20
					standard_of_living < 25
				}
				value = 0.75
			}
		}
		multiply = {
			desc = "SOL_25_30"
			if = { 
				limit = {
					standard_of_living >= 25
					standard_of_living < 30
				}
				value = 0.5
			}
		}
		
		multiply = {
			desc = "LEADER_POPULARITY"
			scope:interest_group = {
				leader = {
					value = popularity
					multiply = 0.0025
					add = 1
					max = 1.25
					min = 0.75
				}
			}
		}		

		multiply = {
			desc = "decentralized_power"	
			value = 1.0
			
			if = {
				limit = { 
					owner = {
						is_country_type = decentralized
					}
				}
				value = 0.0
			}		
		}
		if = {
			limit = {
				scope:interest_group = {
					is_in_government = yes
				}
			}
			multiply = {
				desc = "IN_GOVERNMENT_ATTRACTION"
				value = 1
				add = scope:interest_group.modifier:interest_group_in_government_attraction_mult
				min = 0
			}
		}
	}

	monarch_weight = {
		value = 0.25
	}	
	
	commander_weight = {
		value = 1.0
	}	
	
	noble_chance = {
		value = 0.1
	}
	
	female_commander_chance = {	
		value = 0.0
	}
	
	female_politician_chance = {
		value = 0.0

		if = {
			limit = {
				owner = {
					has_law = law_type:law_women_own_property
				}
				
			}
			add = {
				value = 0.1
			}
		}

		if = {
			limit = {
				owner = {
					has_law = law_type:law_women_in_the_workplace
				}
				
			}
			add = {
				value = 0.1
			}
		}

		if = {
			limit = {
				owner = {
					has_law = law_type:law_womens_suffrage
				}
				
			}
			add = {
				value = 0.1
			}
		}
	}
	
	# The chance that a commander belonging to this IG takes over leadership when it changes
	# scope:character is the most popular commander in the IG
	commander_leader_chance = {
		value = 0.25
		if = {
			limit = {
				exists = scope:character
			}
			multiply = {
				value = 1
				add = scope:character.popularity
				min = 0
			}
		}
	}
}
